emperors_children = {
	damage = 3.0
	health = 8.0
	morale = 3.0
	collateral_damage = 1.00
	war_exhaustion = 0.0
	time = 360
	
	icon_frame = 25
	
	resources = {
		category = armies
		cost = {
			minerals = 550
		}

		upkeep = {
			energy = 5
		}
	}
	
	#prerequisites = { "tech_gene_seed_purification" }
	
	allow = {
		#has_building = "building_emperors_children_fortress_monastery"
		owner = {
			not = { 
				check_variable = {
					which = "emperors_children_count" 
					value = 6
				}
			}
		}
		
	}
	
	on_queued = {
		owner = {
			change_variable = {
				which = "emperors_children_count" 
				value = 1
			}
		}
	}
	
	on_unqueued = {
		owner = {
			change_variable = {
				which = "emperors_children_count" 
				value = -1
			}
		}
	}
	potential = {
		planet = {
			#has_building = "building_emperors_children_fortress_monastery"
		}
		owner = {
			has_country_flag = chaos_empire
		}
	}
}
