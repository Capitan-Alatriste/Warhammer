district_kmenic_quarry = {
	base_buildtime = 600
	is_capped_by_modifier = no

	show_on_uncolonized = {
		uses_district_set = tomb_world
	}

	potential = {
		uses_district_set = tomb_world
		exists = owner
		owner = { has_country_flag = necron_empire }
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 10
		}
	}

	planet_modifier = {
		planet_housing_add = 10
		job_necron_miner_add = 10
	}
	
	triggered_desc = {
		text = job_necron_miner_effect_desc
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 3
			OR = {
				free_jobs < 2
				AND = {
					districts_good_minerals = yes
					free_jobs < 6
				}
				any_owned_pop = {
					is_unemployed = yes
					NOT = { has_citizenship_type = { type = citizenship_organic_trophy } }
					NOR = {
						is_pop_category = specialist
						is_pop_category = ruler
					}
				}
			}
		}
		modifier = {
			weight = 300
			OR = {
				AND = {
					years_passed > 4
					OR = {
						free_jobs < 3
						owner = { has_ascension_perk = ap_arcology_project }
						any_owned_pop = {
							is_unemployed = yes
							NOT = { has_citizenship_type = { type = citizenship_organic_trophy } }
							NOR = {
								is_pop_category = specialist
								is_pop_category = ruler
							}
						}
					}
					owner = {
						has_monthly_income = {
							resource = minerals
							value < 100
						}
					}
					OR = {
						owner = {
							has_monthly_income = {
								resource = food
								value > 0
							}
						}
						num_free_districts = { type = district_farming value = 0 }
					}
				}
				AND = {
					free_jobs < 5
					districts_good_energy = no
					districts_good_food = no
					owner = {
						used_naval_capacity_percent < 0.9
						has_monthly_income = {
							resource = alloys
							value <= 5
						}#Alloy production has shut down which means the current mineral income can't be trusted
					}
				}
			}
		}
		modifier = {
			factor = 1.25
			exists = owner
			owner = {
				is_lithoid_empire = yes
			}
		}
		modifier = {
			factor = 10
			owner = {
				has_deficit = minerals
			}
		}
	}

	ai_resource_production = {
		minerals = 1
		trigger = {
			always = yes
		}
	}
}
