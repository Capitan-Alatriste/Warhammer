district_special_biodome = {
	base_buildtime = 600
	is_capped_by_modifier = no
	
	show_on_uncolonized = {
		uses_district_set = craft_world
	}

	potential = {
		uses_district_set = craft_world
		exists = owner
		owner = { has_country_flag = eldarian }
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 10
		}
	}

	planet_modifier = {
		planet_housing_add = 10
		job_farmer_add = 10
	}

	triggered_desc = {
		text = job_farmer_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 3
			owner = {
				country_uses_food = yes
				has_monthly_income = {
					resource = food
					value < 100
				}
			}
			num_free_districts = { type = district_mining value = 0 }
			num_free_districts = { type = district_generator value = 0 }
			OR = {
				free_jobs < 2
				any_owned_pop = {
					is_unemployed = yes
					NOT = { has_citizenship_type = { type = citizenship_organic_trophy } }
					NOR = {
						is_pop_category = specialist
						is_pop_category = ruler
					}
				}
			}
		}
		modifier = {
			weight = 250
			years_passed > 4
			OR = {
				free_jobs < 3
				any_owned_pop = {
					is_unemployed = yes
					NOT = { has_citizenship_type = { type = citizenship_organic_trophy } }
					NOR = {
						is_pop_category = specialist
						is_pop_category = ruler
					}
				}
			}
			OR = {
				owner = {
					country_uses_food = yes
					has_monthly_income = {
						resource = food
						value < 20
					}
				}
				owner = {
					country_uses_food = no
					has_monthly_income = {
						resource = food
						value < -1
					}
				}
			}#If an AI has matrix'd pops it will need food.. not much food but some.
		}
		modifier = {
			weight = 1
			free_jobs < 5
			owner = {
				has_resource = {
					type = food
					amount < 10
				}
				has_monthly_income = {
					resource = food
					value < 0
				}
			}
		}

		modifier = {
			factor = 10
			owner = {
				has_deficit = food
			}
		}
		modifier = {
			factor = 0
			owner = {
				country_uses_food = no
				NOT = { has_deficit = food }
			}
		}
	}

	ai_resource_production = {
		food = 1
		trigger = {
			always = yes
		}
	}
}
