district_hive_city = {
	base_buildtime = 600
	is_capped_by_modifier = no

	show_on_uncolonized = {
		uses_district_set = hivecity_world
	}

	potential = {
		uses_district_set = hivecity_world
		exists = owner
		owner = { has_country_flag = humanoid }
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 10
		}
	}

	planet_modifier = {
		planet_housing_add = 30
		job_clerk_add = 10
		job_bureaucrat_add = 10
	}

	triggered_desc = {
		text = job_clerk_effect_desc
	}
	
	triggered_desc = {
		text = job_bureaucrat_effect_desc
	}

	ai_weight = {
		weight = 0
		modifier = {
			weight = 5
			free_housing < 10
		}
		modifier = {
			weight = 10
			owner = { empire_sprawl_over_cap > 0 }
		}
		modifier = {
			factor = 2
			owner = { empire_sprawl_over_cap > 10 }
		}
		modifier = {
			factor = 2
			owner = { empire_sprawl_over_cap > 20 }
		}
		modifier = {
			factor = 2
			owner = { empire_sprawl_over_cap > 40 }
		}
	}
}

district_military_manufactoria = {
	base_buildtime = 600
	is_capped_by_modifier = no

	show_on_uncolonized = {
		OR = {
			uses_district_set = hivecity_world
			uses_district_set = fortress_world
		}
	}

	potential = {
		OR = {
			uses_district_set = hivecity_world
			uses_district_set = fortress_world
		}
		exists = owner
		owner = { has_country_flag = humanoid }
	}

	resources = {
		category = planet_districts
		cost = {
			minerals = 1000
		}
		upkeep = {
			energy = 10
		}
	}

	planet_modifier = {
		planet_housing_add = 10
		job_military_artisan_add = 10
	}

	triggered_desc = {
		text = job_military_artisan_effect_desc
	}
	
	ai_weight = {
		weight = 0
		modifier = {
			weight = 750
			owner = {
				has_monthly_income = {
					resource = military_goods
					value < 200
				}
				has_monthly_income = {
					resource = alloys
					value > 99
				}
			}
		}
		modifier = {
			weight = 1000
			owner = {
				has_monthly_income = {
					resource = military_goods
					value < 0
				}
			}
		}
	}
	
	ai_resource_production = {
		military_goods = 1
		trigger = {
			always = yes
		}
	}
}
